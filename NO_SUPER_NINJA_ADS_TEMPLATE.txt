# SUPER_NINJA ADVERTISEMENT REMOVAL PROTOCOL

## When Creating New Pages or Articles:

### 1. CSS TO ADD (in <head> section, before </head>):
```html
    <!-- HIDE SUPER_NINJA ADVERTISEMENTS -->
    <style>
      .ninja-badge,
      .ninja-daytona-banner,
      .ninja-connection-status,
      .ninja-disconnect-overlay,
      .ninja-badge-close,
      /* Additional SuperNinja elements */
      [id*="ninja"],
      [class*="ninja-banner"],
      [class*="ninja-badge"] {
        display: none !important;
        visibility: hidden !important;
        opacity: 0 !important;
        position: absolute !important;
        left: -9999px !important;
        top: -9999px !important;
        pointer-events: none !important;
      }
    </style>
```

### 2. JAVASCRIPT TO ADD (immediately after <body>, before any content):
```html
    <!-- IMMEDIATE SUPER_NINJA REMOVAL - RUNS BEFORE ADS LOAD -->
    <script>
      (function() {
        // Immediate removal - runs synchronously
        function hideNinjaElements() {
          var selectors = [
            '.ninja-badge',
            '.ninja-daytona-banner',
            '.ninja-connection-status',
            '.ninja-disconnect-overlay',
            '[id*="ninja"]',
            '[class*="ninja-banner"]',
            '[class*="ninja-badge"]'
          ];
          selectors.forEach(function(selector) {
            var elements = document.querySelectorAll(selector);
            for (var i = 0; i < elements.length; i++) {
              elements[i].style.display = 'none';
              elements[i].style.visibility = 'hidden';
              elements[i].style.opacity = '0';
              elements[i].remove();
            }
          });
        }
        
        // Run immediately
        hideNinjaElements();
        
        // Run on DOMContentLoaded as backup
        if (document.readyState !== 'loading') {
          hideNinjaElements();
        } else {
          document.addEventListener('DOMContentLoaded', hideNinjaElements);
        }
        
        // Run continuously to catch any dynamically added elements
        setInterval(hideNinjaElements, 1000);
      })();
    </script>
```

### 3. DO NOT ADD THIS SCRIPT TAG (if it appears, remove it):
```html
<script src="https://sites.super.myninja.ai/_assets/ninja-daytona-script.js"></script>
```
